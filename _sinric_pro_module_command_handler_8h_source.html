67 bool SinricProModuleCommandHandler::handleRequest(SinricProRequest &request) {
68  if (strcmp(FSTR_OTA_otaUpdateAvailable, request.action.c_str()) == 0 && _otaUpdateCallbackHandler) {
69  String url = request.request_value[FSTR_OTA_url];
70  int major = request.request_value[FSTR_OTA_version][FSTR_OTA_major];
71  int minor = request.request_value[FSTR_OTA_version][FSTR_OTA_minor];
72  int patch = request.request_value[FSTR_OTA_version][FSTR_OTA_patch];
73  bool forceUpdate = request.request_value[FSTR_OTA_version][FSTR_OTA_forceUpdate] | false;
74  return _otaUpdateCallbackHandler(url, major, minor, patch, forceUpdate);
75  }
76  else if (strcmp(FSTR_SETTINGS_setSetting, request.action.c_str()) == 0 && _setSettingCallbackHandler) {
77  String id = request.request_value[FSTR_SETTINGS_id];
78  String value = request.request_value[FSTR_SETTINGS_value];
79  return _setSettingCallbackHandler(id, value);
80  }
81  else if (strcmp(FSTR_INSIGHTS_health, request.action.c_str()) == 0 && _reportHealthCallbackHandler) {
82  String healthReport = "";
83  bool success = _reportHealthCallbackHandler(healthReport);
84  if (success) {
85  request.response_value[FSTR_INSIGHTS_report] = healthReport;
86  }
87  return success;
88  }
89  else {
90  DEBUG_SINRIC("[SinricProModuleCommandHandler:handleRequest]: action: %s not supported!\r\n", request.action.c_str());
91  }
92  return false;
93 }
