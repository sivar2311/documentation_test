328 void SinricProClass::handleModuleRequest(JsonDocument& requestMessage, interface_t Interface) {
329  DEBUG_SINRIC("[SinricPro.handleModuleScopeRequest()]: handling device sope request\r\n");
330 #ifndef NODEBUG_SINRIC
331  serializeJsonPretty(requestMessage, DEBUG_ESP_PORT);
332 #endif
333 
334  JsonDocument responseMessage = prepareResponse(requestMessage);
335 
336  String action = requestMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_action] | "";
337  JsonObject request_value = requestMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_value];
338  JsonObject response_value = responseMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_value];
339  SinricProRequest request{ action, "", request_value, response_value};
340 
341  bool success = _moduleCommandHandler.handleRequest(request);
342 
343  responseMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_success] = success;
344  responseMessage[FSTR_SINRICPRO_payload].remove(FSTR_SINRICPRO_deviceId);
345  if (!success) {
346  if (responseMessageStr.length() > 0) {
347  responseMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_message] = responseMessageStr;
348  responseMessageStr = "";
349  } else {
350  responseMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_message] = "Module did not handle \"" + action + "\"";
351  }
352  }
353 
354  String responseString;
355  serializeJson(responseMessage, responseString);
356  sendQueue.push(new SinricProMessage(Interface, responseString.c_str()));
357 }

359 void SinricProClass::handleDeviceRequest(JsonDocument& requestMessage, interface_t Interface) {
360  DEBUG_SINRIC("[SinricPro.handleDeviceRequest()]: handling device sope request\r\n");
361 #ifndef NODEBUG_SINRIC
362  serializeJsonPretty(requestMessage, DEBUG_ESP_PORT);
363 #endif
364 
365  JsonDocument responseMessage = prepareResponse(requestMessage);
366 
367  // handle devices
368  bool success = false;
369  const char* deviceId = requestMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_deviceId];
370  String action = requestMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_action] | "";
371  String instance = requestMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_instanceId] | "";
372  JsonObject request_value = requestMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_value];
373  JsonObject response_value = responseMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_value];
374 
375  for (auto& device : devices) {
376  if (device->getDeviceId() == deviceId && success == false) {
377  SinricProRequest request{
378  action,
379  instance,
380  request_value,
381  response_value};
382  success = device->handleRequest(request);
383  responseMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_success] = success;
384  if (!success) {
385  if (responseMessageStr.length() > 0) {
386  responseMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_message] = responseMessageStr;
387  responseMessageStr = "";
388  } else {
389  responseMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_message] = "Device did not handle \"" + action + "\"";
390  }
391  }
392  }
393  }
394 
395  String responseString;
396  serializeJson(responseMessage, responseString);
397  sendQueue.push(new SinricProMessage(Interface, responseString.c_str()));
398 }

400 void SinricProClass::handleReceiveQueue() {
401  if (receiveQueue.size() == 0) return;
402 
403  DEBUG_SINRIC("[SinricPro.handleReceiveQueue()]: %i message(s) in receiveQueue\r\n", receiveQueue.size());
404  while (receiveQueue.size() > 0) {
405  SinricProMessage* rawMessage = receiveQueue.front();
406  receiveQueue.pop();
407  JsonDocument jsonMessage;
408  deserializeJson(jsonMessage, rawMessage->getMessage());
409 
410  bool sigMatch = false;
411 
412  if (strncmp(rawMessage->getMessage(), "{\"timestamp\":", 13) == 0 && strlen(rawMessage->getMessage()) <= 26) {
413  sigMatch = true; // timestamp message has no signature...ignore sigMatch for this!
414  } else {
415  String signature = jsonMessage[FSTR_SINRICPRO_signature][FSTR_SINRICPRO_HMAC] | "";
416  String payload = extractPayload(rawMessage->getMessage());
417  String calculatedSignature = calculateSignature(appSecret.c_str(), payload);
418  sigMatch = (calculatedSignature == signature);
419  }
420 
421  String messageType = jsonMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_type];
422 
423  if (sigMatch) { // signature is valid process message
424  DEBUG_SINRIC("[SinricPro.handleReceiveQueue()]: Signature is valid. Processing message...\r\n");
425  extractTimestamp(jsonMessage);
426  if (messageType == FSTR_SINRICPRO_response) handleResponse(jsonMessage);
427  if (messageType == FSTR_SINRICPRO_request) {
428  String scope = jsonMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_scope] | FSTR_SINRICPRO_device;
429  if (strcmp(FSTR_SINRICPRO_module, scope.c_str()) == 0) {
430  handleModuleRequest(jsonMessage, rawMessage->getInterface());
431  } else {
432  handleDeviceRequest(jsonMessage, rawMessage->getInterface());
433  }
434  };
435  } else {
436  DEBUG_SINRIC("[SinricPro.handleReceiveQueue()]: Signature is invalid! \r\n");
437  if (messageType == FSTR_SINRICPRO_request) handleDeviceRequest(jsonMessage, rawMessage->getInterface());
438  }
439  delete rawMessage;
440  }
441 }

443 void SinricProClass::handleSendQueue() {
444  if (!isConnected()) return;
445  if (!timestamp.getTimestamp()) return;
446  while (sendQueue.size() > 0) {
447  DEBUG_SINRIC("[SinricPro:handleSendQueue()]: %i message(s) in sendQueue\r\n", sendQueue.size());
448  DEBUG_SINRIC("[SinricPro:handleSendQueue()]: Sending message...\r\n");
449 
450  SinricProMessage* rawMessage = sendQueue.front();
451  sendQueue.pop();
452 
453  JsonDocument jsonMessage;
454  deserializeJson(jsonMessage, rawMessage->getMessage());
455  jsonMessage[FSTR_SINRICPRO_payload][FSTR_SINRICPRO_createdAt] = timestamp.getTimestamp();
456  signMessage(appSecret, jsonMessage);
457 
458  String messageStr;
459 
460  serializeJson(jsonMessage, messageStr);
461 #ifndef NODEBUG_SINRIC
462  serializeJsonPretty(jsonMessage, DEBUG_ESP_PORT);
463  Serial.println();
464 #endif
465 
466  switch (rawMessage->getInterface()) {
467  case IF_WEBSOCKET:
468  DEBUG_SINRIC("[SinricPro:handleSendQueue]: Sending to websocket\r\n");
469  _websocketListener.sendMessage(messageStr);
470  break;
471  case IF_UDP:
472  DEBUG_SINRIC("[SinricPro:handleSendQueue]: Sending to UDP\r\n");
473  _udpListener.sendMessage(messageStr);
474  break;
475  default:
476  break;
477  }
478  delete rawMessage;
479  DEBUG_SINRIC("[SinricPro:handleSendQueue()]: message sent.\r\n");
480  }
481 }

596  unsigned long tempTimestamp = 0;
597  // extract timestamp from timestamp message right after websocket connection is established
598  tempTimestamp = message["timestamp"] | 0;
